:: lang_version: neuroscript@0.5.0
:: file_version: 2.0

command
    :: description: Lists available tools with formatted signatures.
    :: purpose: Provides a clean, readable list of all registered tools.

    set toolList = tool.Meta.ListTools()
    emit "Available Tools:"
    emit "----------------"

    for each toolSpec in toolList
        set signature = ""
        set params = toolSpec["args"]

        if params != nil and len(params) > 0
            set paramStrings = []
            for each param in params
                set pName = param["name"]
                set pType = param["type"]
                set paramStrings = tool.list.Append(paramStrings, pName + ": " + pType)
            endfor
            set signature = tool.str.Join(paramStrings, ", ")
        endif

        set fullName = toolSpec["fullname"]
        set returnType = toolSpec["returnType"]

        // Format: tool.group.Name(param1: type, param2: type) -> returnType
        set finalOutput = fullName + "(" + signature + ") -> " + returnType
        emit finalOutput
    endfor
endcommand

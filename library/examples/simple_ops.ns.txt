func main means
  :: description: Demonstrates basic operations and text output.
  set first_name = "Neuro"
  set last_name = "Script"
  set number_a = 10
  set number_b = 5
  set sum_result = number_a + number_b
  set full_name = first_name + " " + last_name
  call emit("Starting simple demo...")
  call emit("Full Name: " + full_name)
  call emit("Calculation: " + number_a + " + " + number_b + " = " + sum_result)
  call emit("Demo finished.")
endfunc

func main() means
  :: purpose: Main entry point for the script. Walks the current directory and prints file paths.
  :: returns: None explicitly, but prints to stdout.
  :: caveats: This version assumes tool.FS.Walk returns a list of maps on success and relies on on_error for tool call failures, as $LAST_ERROR, 'is list', and 'typeof' are not in the current script_spec.md.

  on_error means
    emit "An error occurred during script execution."
    fail
  endon

  set allEntries = tool.FS.Walk(".")

  emit "Files found:"

  for each entry in allEntries
    if entry["is_dir"] == false
      emit "- " + entry["path_relative"]
    endif
  endfor

  emit "Directory scan complete."

endfunc

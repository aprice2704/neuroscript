                    DEFINE PROCEDURE DemoFeatures(input_topic)
                    COMMENT:
                      PURPOSE: Demonstrates LLM call, tool use, and basic logic.
                      INPUTS: - input_topic: A string topic (e.g., "NeuroScript")
                      OUTPUT: A generated description or a default message.
                    ENDCOMMENT
                    
                      SET topic = input_topic
                      EMIT "Analyzing topic: " + topic
                    
                      # Tool Call Example
                      CALL TOOL.StringLength(topic)
                      SET topic_length = LAST
                    
                      # LLM Call Example (using EVAL for interpolation)
                      SET prompt = EVAL("Generate a one-sentence description for: {{topic}} ({{topic_length}} chars)")
                      CALL LLM(prompt)
                      SET llm_description = LAST
                    
                      # Conditional Logic & Procedure Call Example
                      IF llm_description != "" THEN
                        # Assuming 'LogResult' is another defined procedure
                        CALL LogResult(topic, llm_description)
                        RETURN llm_description
                      ELSE
                        RETURN "Could not generate description."
                      ENDBLOCK
                    
                    END
